int num1;

int mod(int x, int y){
	return(x - x / y * y);
}

int gcd(int x, int y){
	if(y == 0)
		return(x);
	return(gcd(y, mod(x, y)));
}
void main(){
	int num2;
	printf("input two numbers(>0):");
	scanf(num1, num2);
	if(num1 <= 0){
		printf("input error.");
		return;
	}
	if(num2 <= 0){
		printf("input error.");
		return;
	}
	printf(gcd(num1, num2));
}
